278bool DynamicCollisionComponent::borderCollision()
279{
280 if (position->position.x < 0)
281 {
282 position->position.x = 0;
283 position->velocity.x = 0;
284
285 collisionXcorner1 = true;
286 }
287 if (position->position.y < 0)
288 {
289 position->position.y = 0;
290 position->sign.y = 1;
291
292 collisionYcorner1 = true;
293 }
294 if (position->position.x + position->width > Map::cols * TILE_SIZE)
295 {
296 position->position.x = static_cast<float>(Map::cols * TILE_SIZE - position->width);
297 position->velocity.x = 0;
298
299 collisionXcorner2 = true;
300 }
301 if (position->position.y + position->height > Map::rows * TILE_SIZE)
302 {
303 position->position.y = static_cast<float>(Map::rows * TILE_SIZE - position->height);
304
305 collisionYcorner2 = true;
306 }
307
308 return collisionXcorner1 || collisionXcorner2 || collisionYcorner1 || collisionYcorner2;
